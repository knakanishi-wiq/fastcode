  <files>tests/test_retriever_fts5.py, fastcode/retriever.py</files>
  <behavior>
    _keyword_search(query, top_k, repo_filter) must:
    - Open (or reuse) sqlite3 connection via init_db(db_path)
    - Execute: SELECT c.source_path, c.content, fts.rank
        FROM chunks_fts fts JOIN chunks c ON fts.rowid = c.id
        WHERE chunks_fts MATCH ? AND c.source_path LIKE ?
        ORDER BY fts.rank
        LIMIT ?
      where MATCH token is AND-joined quoted terms from the query string,
      and LIKE pattern is '{repo_path}/%'
    - Return List[Tuple[Dict[str, Any], float]] — each dict has at minimum
      "source_path" and "content" keys; float is abs(fts.rank) (FTS5 rank is negative)
    - Return [] when no chunks match (empty corpus or no match)
    - repo_filter: List[str] of repo_path prefixes (one element typical);
      if empty or None, no LIKE filter applied (search across all repos)

    Cases:
    - query="hello", repo_filter=["/repos/myrepo"] → returns chunks from myrepo containing "hello"
    - query="hello", repo_filter=[] → returns chunks from all repos
    - query="hello", repo_filter=["/repos/otherrepo"] → returns [] (myrepo not in scope)
    - corpus empty → returns []
    - top_k=2 with 10 matches → returns exactly 2 results
  </behavior>

       ```python
       def _keyword_search(self, query: str, top_k: int = 10,
                           repo_filter: Optional[List[str]] = None
                           ) -> List[Tuple[Dict[str, Any], float]]:
           conn = self._get_db_conn()
           if conn is None or not query.strip():
               return []
           terms = query.strip().split()
           match_expr = " ".join(f'"{t}"' for t in terms)
           if repo_filter:
               sql = (
                   "SELECT c.source_path, c.content, fts.rank "
                   "FROM chunks_fts fts JOIN chunks c ON fts.rowid = c.id "
                   "WHERE chunks_fts MATCH ? AND c.source_path LIKE ? "
                   "ORDER BY fts.rank LIMIT ?"
               )
               like_pat = f"{repo_filter[0]}/%"
               rows = conn.execute(sql, (match_expr, like_pat, top_k)).fetchall()
           else:
               sql = (
                   "SELECT c.source_path, c.content, fts.rank "
                   "FROM chunks_fts fts JOIN chunks c ON fts.rowid = c.id "
                   "WHERE chunks_fts MATCH ? "
                   "ORDER BY fts.rank LIMIT ?"
               )
               rows = conn.execute(sql, (match_expr, top_k)).fetchall()
           results = []
           for source_path, content, rank in rows:
               metadata = {
                   "source_path": source_path,
                   "file_path": source_path,
                   "relative_path": source_path,
                   "content": content,
                   "code": content,
                   "id": source_path,
                   "name": os.path.basename(source_path),
                   "type": "file",
                   "repo_name": None,
               }
               results.append((metadata, abs(rank)))
           self.logger.debug(f"FTS5 keyword search found {len(results)} results")
           return results
       ```
